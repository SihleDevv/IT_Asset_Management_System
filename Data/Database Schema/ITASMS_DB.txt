/* ============================================================
   IDENTITY TABLES
============================================================ */

CREATE TABLE dbo.AspNetRoles(
    Id nvarchar(450) NOT NULL PRIMARY KEY,
    Name nvarchar(256) NULL,
    NormalizedName nvarchar(256) NULL,
    ConcurrencyStamp nvarchar(max) NULL
);

CREATE TABLE dbo.AspNetUsers(
    Id nvarchar(450) NOT NULL PRIMARY KEY,
    FullName nvarchar(max) NULL,
    Department nvarchar(max) NULL,
    CreatedDate datetime2 NOT NULL,
    IsActive bit NOT NULL,
    UserName nvarchar(256) NULL,
    NormalizedUserName nvarchar(256) NULL,
    Email nvarchar(256) NULL,
    NormalizedEmail nvarchar(256) NULL,
    EmailConfirmed bit NOT NULL,
    PasswordHash nvarchar(max) NULL,
    SecurityStamp nvarchar(max) NULL,
    ConcurrencyStamp nvarchar(max) NULL,
    PhoneNumber nvarchar(max) NULL,
    PhoneNumberConfirmed bit NOT NULL,
    TwoFactorEnabled bit NOT NULL,
    LockoutEnd datetimeoffset NULL,
    LockoutEnabled bit NOT NULL,
    AccessFailedCount int NOT NULL,
    PasswordChangedDate datetime2 NULL,
    MustChangePassword bit NOT NULL DEFAULT 0
);

CREATE TABLE dbo.AspNetUserRoles(
    UserId nvarchar(450) NOT NULL,
    RoleId nvarchar(450) NOT NULL,
    CONSTRAINT PK_AspNetUserRoles PRIMARY KEY (UserId, RoleId)
);

CREATE TABLE dbo.AspNetUserClaims(
    Id int IDENTITY(1,1) PRIMARY KEY,
    UserId nvarchar(450) NOT NULL,
    ClaimType nvarchar(max),
    ClaimValue nvarchar(max)
);

CREATE TABLE dbo.AspNetRoleClaims(
    Id int IDENTITY(1,1) PRIMARY KEY,
    RoleId nvarchar(450) NOT NULL,
    ClaimType nvarchar(max),
    ClaimValue nvarchar(max)
);

CREATE TABLE dbo.AspNetUserLogins(
    LoginProvider nvarchar(450) NOT NULL,
    ProviderKey nvarchar(450) NOT NULL,
    ProviderDisplayName nvarchar(max),
    UserId nvarchar(450) NOT NULL,
    CONSTRAINT PK_AspNetUserLogins PRIMARY KEY (LoginProvider, ProviderKey)
);

CREATE TABLE dbo.AspNetUserTokens(
    UserId nvarchar(450) NOT NULL,
    LoginProvider nvarchar(450) NOT NULL,
    Name nvarchar(450) NOT NULL,
    Value nvarchar(max),
    CONSTRAINT PK_AspNetUserTokens PRIMARY KEY (UserId, LoginProvider, Name)
);

CREATE TABLE dbo.Assets(
    Id int IDENTITY(1,1) PRIMARY KEY,
    AssetTag nvarchar(100) NOT NULL,
    AssetName nvarchar(200) NOT NULL,
    AssetType nvarchar(50) NOT NULL,
    Brand nvarchar(100) NOT NULL,
    Model nvarchar(100) NOT NULL,
    SerialNumber nvarchar(100) NOT NULL,
    PurchaseDate datetime2 NOT NULL,
    PurchasePrice decimal(18,2),
    Vendor nvarchar(100) NOT NULL,
    WarrantyExpiryDate datetime2,
    Location nvarchar(200) NOT NULL,
    Status nvarchar(50) NOT NULL,
    Notes nvarchar(500) NOT NULL,
    CreatedDate datetime2 NOT NULL,
    CreatedBy nvarchar(max),
    ModifiedDate datetime2,
    ModifiedBy nvarchar(max),
    VendorType nvarchar(100) NOT NULL DEFAULT ''
);

CREATE TABLE dbo.Servers(
    Id int PRIMARY KEY,
    IPAddress nvarchar(100) NOT NULL,
    ServerType nvarchar(50) NOT NULL,
    Purpose nvarchar(100) NOT NULL,
    Processor nvarchar(50) NOT NULL,
    RAM int NOT NULL,
    Storage int NOT NULL,
    OperatingSystem nvarchar(50) NOT NULL,
    BackupComments nvarchar(500) NOT NULL DEFAULT '',
    BackupRequired bit NOT NULL DEFAULT 0,
    ProjectManagerName nvarchar(200) NOT NULL DEFAULT ''
);

CREATE TABLE dbo.Computers(
    Id int PRIMARY KEY,
    Processor nvarchar(50) NOT NULL,
    RAM int NOT NULL,
    Storage int NOT NULL,
    OperatingSystem nvarchar(50) NOT NULL,
    AssignedTo nvarchar(200) NOT NULL
);

CREATE TABLE dbo.Applications(
    Id int PRIMARY KEY,
    Version nvarchar(100) NOT NULL,
    Category nvarchar(50) NOT NULL,
    Description nvarchar(500) NOT NULL,
    RequiresLicense bit NOT NULL,
    LicenseKey nvarchar(200) NOT NULL,
    LicenseType nvarchar(50) NOT NULL,
    TotalLicenses int,
    UsedLicenses int,
    LicenseExpiryDate datetime2,
    ApplicationOwner nvarchar(200) NOT NULL DEFAULT '',
    BusinessUnit nvarchar(100) NOT NULL DEFAULT '',
    LicenseHolder nvarchar(50) NOT NULL DEFAULT ''
);

CREATE TABLE dbo.AssetServerApplications(
